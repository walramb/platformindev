visualizekeyboard = (mappings) ->
  
  output = "<div class='keyboardlayout'>"
  for row in keyboardrows
    output += "<div>"
    for key in row
      if mappings[key]?
        text=mappings[key]
        output += "<span class='highlight' title='#{text}'>#{key}</span>"
      else
        output += "<span>#{key}</span>"
    output += "</div>"
  return output


root = exports ? this

root.keyboardlayout = {}
root.keyboardlayout.visualize = visualizekeyboard
